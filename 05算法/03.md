### 寻找数组重复的数

找到数组中重复的一个数并返回。

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var arr = [1, 2, 3, 4, 4, 5, 8, 6, 3, 2, 1];

var findRepeatNumber = function (nums) {
    var len = nums.length;
    if (len >= 2 && len <= 100000) {
         for (var i = 0; i < nums.length; i++) {
            for (var j = i + 1; j < nums.length; j++) {
                if (nums[i] == nums[j]) {
                    return nums[j]
                }
            }
        }
    } else {
       return
    }
};

findRepeatNumber(arr)
```
